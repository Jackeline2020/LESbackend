CREATE TABLE CUPOM(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR(30),
    PORCENTAGEM_DESCONTO        NUMBER(10,4),
    CONSTRAINT CUPOM_ID PRIMARY KEY (ID)
);

CREATE TABLE ESTADO(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR2(100),
    SIGLA                       VARCHAR2(2),
    CONSTRAINT ESTADO_ID PRIMARY KEY (ID)
);

CREATE TABLE CIDADE(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR2(100),
    ESTADO_ID                   NUMBER(8),
    CONSTRAINT CIDADE_ID PRIMARY KEY (ID),
    CONSTRAINT FK_ESTADO_CIDADE FOREIGN KEY (ESTADO_ID) REFERENCES ESTADO(ID)
);

CREATE TABLE PESSOA(
    ID                          NUMBER(8)                           NOT NULL,
    TIPO                        VARCHAR2(20)                        NOT NULL,
    CONSTRAINT PESSOA_ID PRIMARY KEY (ID)
);

CREATE TABLE TIPO_CLIENTE(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR2(100),
    DESCRICAO                   VARCHAR2(200),
    CONSTRAINT TIPO_CLIENTE_ID PRIMARY KEY (ID)    
);

CREATE TABLE CLIENTE(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR2(100),
    EMAIL                       VARCHAR2(100),
    SENHA                       VARCHAR2(40),
    TIPO_CLIENTE_ID             NUMBER(8),
    CONSTRAINT CLIENTE_ID PRIMARY KEY (ID),
    CONSTRAINT FK_TIPO_CLIENTE FOREIGN KEY (TIPO_CLIENTE_ID) REFERENCES TIPO_CLIENTE (ID)
);

CREATE TABLE TIPO_DOCUMENTO(
    ID                          NUMBER(8)                           NOT NULL,
    NOME                        VARCHAR2(100),
    DESCRICAO                   VARCHAR2(200),
    CONSTRAINT TIPO_DOCUMENTO_ID PRIMARY KEY (ID)
);

CREATE TABLE DOCUMENTO(
    ID                          NUMBER(8)                           NOT NULL,
    NUMERO_DOCUMENTO            VARCHAR(30),
    VALIDADE                    DATE ,
    TIPO_DOCUMENTO_ID           NUMBER(8),
    CLIENTE_ID                  NUMBER(8),
    CONSTRAINT DOCUMENTO_ID PRIMARY KEY (ID),
    CONSTRAINT FK_TIPO_DOCUMENTO FOREIGN KEY (TIPO_DOCUMENTO_ID) REFERENCES TIPO_DOCUMENTO (ID),
    CONSTRAINT FK_CLIENTE_DOCUMENTO FOREIGN KEY (CLIENTE_ID) REFERENCES CLIENTE (ID)
);

CREATE TABLE TIPO_ENDERECO(
    ID                          NUMBER(8)                           NOT NULL,
    TIPO                        VARCHAR2(50),
    CONSTRAINT TIPO_ENDERECO_ID PRIMARY KEY (ID)
);

CREATE TABLE ENDERECO(
    ID                          NUMBER(8)                           NOT NULL,
    CEP                         VARCHAR2(8),
    LOGRADOURO                  VARCHAR2(50),
    NUMERO                      VARCHAR2(5),
    BAIRRO                      VARCHAR2(80),
    COMPLEMENTO                 VARCHAR2(30),
    CIDADE_ID                   NUMBER(8),
    TIPO_ENDERECO_ID            NUMBER(8),
    CLIENTE_ID                  NUMBER(8),
    CONSTRAINT ENDERECO_ID PRIMARY KEY (ID),
    CONSTRAINT FK_TIPO_ENDERECO_ID FOREIGN KEY (TIPO_ENDERECO_ID) REFERENCES TIPO_ENDERECO (ID),
    CONSTRAINT FK_CLIENTE_ENDERECO FOREIGN KEY (CLIENTE_ID) REFERENCES CLIENTE (ID)
);

CREATE TABLE PEDIDO(
    ID                          NUMBER(8)                           NOT NULL,
    STATUS_PEDIDO               VARCHAR2(80),
    DATA                        DATE ,
    FORMA_PAGAMENTO             VARCHAR2(50),
    VALOR                       NUMBER(10, 4),
    TIPO_ENTREGA                VARCHAR2(50),
    VALOR_ENTREGA               NUMBER(10, 4),
    CUPOM_ID                    NUMBER(8),
    ENDERECO_ID                 NUMBER(8),
    CLIENTE_ID                  NUMBER(8),
    CONSTRAINT PEDIDO_ID PRIMARY KEY (ID),
    CONSTRAINT FK_CUPOM_PEDIDO FOREIGN KEY (CUPOM_ID) REFERENCES CUPOM (ID),
    CONSTRAINT FK_ENDERECO_PEDIDO FOREIGN KEY (ENDERECO_ID) REFERENCES ENDERECO (ID),
    CONSTRAINT FK_CLIENTE_PEDIDO FOREIGN KEY (CLIENTE_ID) REFERENCES CLIENTE (ID)
);

CREATE TABLE PRODUTO(
    ID                          NUMBER(8)                           NOT NULL,
    DESCRICAO                   VARCHAR2(200),
    VALOR                       NUMBER(10, 4),
    MARCA                       VARCHAR2(50),
    MODELO                      VARCHAR2(80),
    CONTEUDO_EMBALAGEM          VARCHAR2(100),
    GARANTIA                    VARCHAR2(40),
    PESO                        VARCHAR2(40),
    CONSTRAINT PRODUTO_ID PRIMARY KEY (ID)
);

CREATE TABLE PEDIDO_PRODUTO(
    PEDIDO_ID                   NUMBER(8),
    PRODUTO_ID                  NUMBER(8),
    CONSTRAINT FK_PEDIDO_PRODUTO FOREIGN KEY (PEDIDO_ID) REFERENCES PEDIDO (ID),
    CONSTRAINT FK_PRODUTO_PEDIDO FOREIGN KEY (PRODUTO_ID) REFERENCES PRODUTO (ID)
);

CREATE TABLE CONTATO(
    ID                          NUMBER(8)                           NOT NULL,
    DDD                         VARCHAR2(2),
    NUMERO                      VARCHAR2(9),
    CLIENTE_ID                  NUMBER(8),
    CONSTRAINT CONTATO_ID PRIMARY KEY (ID),
    CONSTRAINT FK_CLIENTE_CONTATO FOREIGN KEY (CLIENTE_ID) REFERENCES CLIENTE (ID)
);
